Sidebar collapsing example This example page shows how to collapse sidebar.

Tip!

In this page the sidebar comes colapsed by using the component adm:sidebar with attributecollapsed=true

You can also toggle it by adding/removing the css class sidebar-collapse from page body.

 
               <ui:composition xmlns="http://www.w3.org/1999/xhtml"
                xmlns:ui="http://java.sun.com/jsf/facelets"
                xmlns:p="http://primefaces.org/ui"
                xmlns:adm="http://github.com/adminfaces"
                template="/WEB-INF/templates/template.xhtml">


    <ui:define name="title">
           Sidebar collapsing example
    </ui:define>

    <ui:define name="description">
        This example page shows how to collapse sidebar.
    </ui:define>

   <ui:define name="body">
       <adm:breadCrumb link="/pages/layout/sidebar-collapsed" title="Sidebar"/>
       <adm:sidebar collapsed="true"/>
       <div class="box box-primary">
                <div class="box-header with-border">
                    <div class="callout callout-info">
                        <h4>Tip!</h4>
                        <p>In this page the sidebar comes colapsed by using the component adm:sidebar with attribute<code>collapsed=true</code></p>
                        <p>You can also toggle it by adding/removing the css class <code>sidebar-collapse</code> from <code>page body</code>.</p>
                    </div>
                    <p:button styleClass="btn-primary" value="Toggle" onclick="toggleSidebar(); return false"/>
               </div>
       </div>
   </ui:define>
</ui:composition>
